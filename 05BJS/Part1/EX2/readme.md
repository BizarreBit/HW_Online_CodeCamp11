1. ให้ระบุค่าของ a, b , c และ d หลังจบ statement ทั้งสามบรรทัด
let a = 1, b = 1;
>> a คือ 1, b คือ 1, c และ d ยังไม่ถูกประกาศตัวแปร
let c = ++a;
>> a คือ 2, b คือ 1, c คือ 2, d ยังไม่ถูกประกาศตัวแปร
let d = b++;
>> a คือ 2, b คือ 2, c คือ 2, d คือ 1
